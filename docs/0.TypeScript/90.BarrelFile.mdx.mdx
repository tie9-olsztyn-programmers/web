# Barrel file

:::tip Co to jest *barrel file*?
- to zwyky plik (najczciej `index.ts`),
- nic sam nie robi,

## Struktura projektu

```
src/
  app.ts
  students/
    addStudent.ts
    listStudents.ts
    countStudents.ts
    index.ts         <-- barrel file
  types/
    student.ts
    index.ts         <-- barrel file (opcjonalnie)
  data/
    students.data.ts
  utils/

```

### `data`

`students.data.ts`
```ts
import { Student } from "../types";

export let students: Student[] = [];

```

### `students`

`addStudent.ts`
```ts
import { students } from "../data/students.data";
import { Student } from "../types";

export function addStudent(name: string, age: number, grade: number) {
    const newStudent: Student = {
        id: Date.now().toString(),
        name,
        age,
        grade
    };

    students.push(newStudent);
    return newStudent;
}
```

`countStudents.ts`
```ts
import { students } from "../data/students.data";

export function countStudents() {
    return students.length;
}

```

`listStudents.ts`
```ts
import { students } from "../data/students.data";

export function listStudents() {
    return students;
}
```

`index.ts`
```ts
// to jest barrel
export * from "./addStudent";
export * from "./listStudents";
export * from "./countStudents";

```
